    /*
     * The Pentium Pro and later support global PDE/PTE entries,

     */
    pmap_global_ext = (base_cpuid.feature_flags & CPUF_PAGE_GLOBAL_EXT)
        ? INTEL_PTE_GLOBAL : 0;
